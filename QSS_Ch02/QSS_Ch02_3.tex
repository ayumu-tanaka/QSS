\section{2.3
因果効果と反事実}\label{ux56e0ux679cux52b9ux679cux3068ux53cdux4e8bux5b9f}

\begin{frame}{因果関係の定義}
\phantomsection\label{ux56e0ux679cux95a2ux4fc2ux306eux5b9aux7fa9}
\begin{itemize}
\tightlist
\item
  \textbf{因果関係}:
  ある変数（処置）の変化が別の変数（結果）の変化を引き起こすこと。
\item
  \textbf{反事実} (Counterfactual):
  「もし処置が行われなかったら、結果はどうなっていたか？」という、現実には観察できないシナリオ。
\item
  \textbf{因果効果}: 「処置が行われた場合の結果
  (\(Y(1)\))」と「反事実の結果 (\(Y(0)\))」の差。

  \begin{itemize}
  \tightlist
  \item
    \(\text{因果効果} = \tau_i = Y_i(1) - Y_i(0)\)
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{因果推論の根本問題}
\phantomsection\label{ux56e0ux679cux63a8ux8ad6ux306eux6839ux672cux554fux984c}
\begin{itemize}
\tightlist
\item
  \textbf{根本問題}:
  同じ個人について、処置を受けた場合と受けなかった場合の両方の結果を同時に観察することは不可能。
\item
  解決策：個人の効果ではなく、グループ全体の\textbf{平均因果効果 (ATE)}
  を推定することを目指す。

  \begin{itemize}
  \tightlist
  \item
    \textbf{標本平均因果効果 (SATE)}: 標本における \(Y(1)\) の平均と
    \(Y(0)\) の平均の差。
  \end{itemize}
\end{itemize}
\end{frame}

\section{2.4 ランダム化比較試験
(RCT)}\label{ux30e9ux30f3ux30c0ux30e0ux5316ux6bd4ux8f03ux8a66ux9a13-rct}

\begin{frame}{社会的圧力と投票率の実験}
\phantomsection\label{ux793eux4f1aux7684ux5727ux529bux3068ux6295ux7968ux7387ux306eux5b9fux9a13}
\begin{itemize}
\tightlist
\item
  \textbf{問い}:
  「あなたの投票状況を近所に知らせる」というメッセージは投票率を上げるか？
\item
  \textbf{実験デザイン}:

  \begin{itemize}
  \tightlist
  \item
    多数の有権者を、異なるメッセージを送るグループに\textbf{無作為（ランダム）}に割り当てる。
  \item
    「無作為化」により、各グループの特徴が平均的に等しくなるため、グループ間の差を因果効果として解釈できる。
  \end{itemize}
\end{itemize}
\end{frame}

\section{2.5 まとめ}\label{ux307eux3068ux3081}

\begin{frame}[fragile]{このセクションのまとめ}
\phantomsection\label{ux3053ux306eux30bbux30afux30b7ux30e7ux30f3ux306eux307eux3068ux3081}
\begin{itemize}
\tightlist
\item
  \textbf{因果効果}: \(Y(1) - Y(0)\)
  で定義されるが、片方は常に観察できない。
\item
  \textbf{ランダム化比較試験 (RCT)}:

  \begin{itemize}
  \tightlist
  \item
    無作為割り当てによって「処置が行われなかった場合の世界（反事実）」を代表する対照群（Control
    Group）を作り出す。
  \item
    社会的圧力の実験では、Neighbors
    グループにおいて非常に強力な因果効果が確認された。
  \end{itemize}
\item
  \textbf{Rの操作}:

  \begin{itemize}
  \tightlist
  \item
    \texttt{tapply()} を使ったグループごとの統計量の一括計算。
  \item
    \texttt{barplot()} を用いたカテゴリ別の可視化。
  \item
    \texttt{abline()} による基準線の追加。
  \end{itemize}
\end{itemize}
\end{frame}
