\section{6.5 まとめ}\label{ux307eux3068ux3081}}

\begin{frame}[fragile]{第6章のまとめ}
\protect\hypertarget{ux7b2c6ux7ae0ux306eux307eux3068ux3081}{}
\begin{itemize}
\tightlist
\item
  \textbf{不確実性}:
  社会科学のデータには常に偶然が伴う。確率を用いることで、その不確実性を数学的に記述できる。
\item
  \textbf{確率の法則}:

  \begin{itemize}
  \tightlist
  \item
    \textbf{条件付き確率}: 特定の条件下での発生確率。
  \item
    \textbf{独立性}: 片方の出来事がもう片方に影響しない状態。
  \end{itemize}
\item
  \textbf{大標本定理}:

  \begin{itemize}
  \tightlist
  \item
    \textbf{大数の法則}:
    標本を増やせば、推測はより正確（母平均に近く）になる。
  \item
    \textbf{中心極限定理}:
    サンプルサイズが大きければ、標本平均の分布は予測可能（正規分布）になる。これが推測統計の土台となる。
  \end{itemize}
\item
  \textbf{Rの操作}: \texttt{sample()}, \texttt{dbinom()},
  \texttt{pnorm()}, \texttt{runif()}, \texttt{curve()},
  \texttt{hist()}。
\end{itemize}
\end{frame}
