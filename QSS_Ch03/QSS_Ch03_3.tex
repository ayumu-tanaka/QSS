\section{3.3
1変量の分布をビジュアル化する}\label{ux5909ux91cfux306eux5206ux5e03ux3092ux30d3ux30b8ux30e5ux30a2ux30ebux5316ux3059ux308b}

\begin{frame}{データの可視化の重要性}
\phantomsection\label{ux30c7ux30fcux30bfux306eux53efux8996ux5316ux306eux91cdux8981ux6027}
\begin{itemize}
\tightlist
\item
  数値の羅列（平均や標準偏差など）だけでは、データの全体像を把握しきれない。
\item
  グラフを用いることで、以下を確認できる：

  \begin{itemize}
  \tightlist
  \item
    分布の形状（対称性、偏り）
  \item
    外れ値の存在
  \item
    グループ間の違い
  \end{itemize}
\item
  変数の種類（カテゴリ変数か数値変数か）によって、適切なグラフを選択する必要がある。
\end{itemize}
\end{frame}

\section{3.3.4 まとめ}\label{ux307eux3068ux3081}

\begin{frame}[fragile]{このセクションのまとめ}
\phantomsection\label{ux3053ux306eux30bbux30afux30b7ux30e7ux30f3ux306eux307eux3068ux3081}
\begin{itemize}
\tightlist
\item
  \textbf{棒グラフ}: カテゴリ変数に適しており、比率や度数を比較する。
\item
  \textbf{ヒストグラム}:
  数値変数の分布の形（山がどこにあるかなど）を確認する。
\item
  \textbf{箱ひげ図}: 複数のグループ間で分布を比較する際に非常に強力。
\item
  \textbf{グラフの調整}:

  \begin{itemize}
  \tightlist
  \item
    タイトル (\texttt{main}) や軸ラベル (\texttt{xlab}, \texttt{ylab})
    を適切につける。
  \item
    色や線 (\texttt{abline})、テキスト (\texttt{text})
    を追加して情報を強調する。
  \end{itemize}
\item
  \textbf{Rの操作}: \texttt{barplot()}, \texttt{hist()},
  \texttt{boxplot()}, \texttt{abline()}, \texttt{text()}。
\end{itemize}
\end{frame}
