\section{3.5 まとめ}\label{ux307eux3068ux3081}}

\begin{frame}[fragile]{第3章のまとめ}
\protect\hypertarget{ux7b2c3ux7ae0ux306eux307eux3068ux3081}{}
\begin{itemize}
\tightlist
\item
  \textbf{測定}:
  複雑な概念（被害、政治的立場など）を数値やカテゴリとして捉える。
\item
  \textbf{視覚化}:
  棒グラフ、ヒストグラム、箱ひげ図、散布図を使い分ける。
\item
  \textbf{欠損値}: \texttt{NA} の適切な処理が必要。
\item
  \textbf{多変量関係}:
  相関関係やクラスタリング（k-means法）を用いて、データの構造を明らかにする。
\item
  Rの操作: \texttt{table()}, \texttt{hist()}, \texttt{boxplot()},
  \texttt{cor()}, \texttt{kmeans()} などを習得した。
\end{itemize}
\end{frame}
