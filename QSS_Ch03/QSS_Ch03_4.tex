\section{3.4 標本調査 (Survey
Sampling)}\label{ux6a19ux672cux8abfux67fb-survey-sampling}

\begin{frame}{標本調査の目的}
\phantomsection\label{ux6a19ux672cux8abfux67fbux306eux76eeux7684}
\begin{itemize}
\tightlist
\item
  \textbf{母集団} (Population)
  全体を調べるのが困難な場合、その一部である\textbf{標本} (Sample)
  を抽出して調査する。
\item
  \textbf{無作為抽出} (Random Sampling):
  母集団の全ての個体が選ばれる確率を等しくすることで、標本から母集団の特徴を正確に推測（推定）できるようにする。
\item
  社会科学における課題：

  \begin{itemize}
  \tightlist
  \item
    抽出の偏り
  \item
    無回答バイアス
  \item
    虚偽の回答
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[fragile]{項目無回答 (Item Nonresponse)}

\begin{frame}[fragile]{敏感な質問への対処 (List Experiment)}
\phantomsection\label{ux654fux611fux306aux8ceaux554fux3078ux306eux5bfeux51e6-list-experiment}
\begin{itemize}
\tightlist
\item
  直接聞きにくい質問（例：武装勢力への支持）に対し、間接的に答えてもらう手法。
\item
  \textbf{リスト実験}:

  \begin{itemize}
  \tightlist
  \item
    統制群：無害な項目のリストを渡し、当てはまる「数」だけを答えてもらう。
  \item
    処置群：無害な項目に「敏感な項目」を1つ加えたリストを渡し、同じく「数」を答えてもらう。
  \item
    \textbf{差}が、敏感な項目に「はい」と答えた人の割合の推定値になる。
  \end{itemize}
\end{itemize}

\scriptsize

\CommentTok{\# list.response: 当てはまる項目の回答数}

\section{3.4.3 まとめ}\label{ux307eux3068ux3081}

\begin{frame}[fragile]{このセクションのまとめ}
\phantomsection\label{ux3053ux306eux30bbux30afux30b7ux30e7ux30f3ux306eux307eux3068ux3081}
\begin{itemize}
\tightlist
\item
  \textbf{無作為抽出}: 母集団を代表する標本を得るための強力な方法。
\item
  \textbf{バランス確認}:
  抽出されたグループとそうでないグループで、背景属性に差がないか（共変量のバランス）を箱ひげ図などで確認することが重要。
\item
  \textbf{バイアスへの警戒}:

  \begin{itemize}
  \tightlist
  \item
    \textbf{無回答}:
    欠損値がランダムに発生しているか、特定のグループに偏っていないかを確認する。
  \item
    \textbf{リスト実験}:
    直接質問では得られない「本音」を統計的に推計する工夫。
  \end{itemize}
\item
  \textbf{Rの操作}: \texttt{is.na()} による欠損値の判定、\texttt{log()}
  による対数変換。
\end{itemize}
\end{frame}
