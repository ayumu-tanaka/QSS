\section{3.7 クラスター化
(Clustering)}\label{ux30afux30e9ux30b9ux30bfux30fcux5316-clustering}

\begin{frame}{クラスター分析の目的}
\phantomsection\label{ux30afux30e9ux30b9ux30bfux30fcux5206ux6790ux306eux76eeux7684}
\begin{itemize}
\tightlist
\item
  \textbf{クラスター分析}:
  多数のデータの中から、互いに似た特徴を持つグループ（クラスター）を自動的に見つけ出す手法。
\item
  \textbf{教師なし学習}:
  正解（ラベル）があらかじめ与えられていない状態で、データの構造を探索する。
\item
  社会科学における例：

  \begin{itemize}
  \tightlist
  \item
    投票行動が似ている議員のグループ化。
  \item
    似たような意識を持つ回答者のセグメンテーション。
  \end{itemize}
\item
  今回は、代表的な手法である \textbf{k-means 法}（k平均法）を学びます。
\end{itemize}
\end{frame}

\section{3.7.3 k-means 法 (k-Means
Algorithm)}\label{k-means-ux6cd5-k-means-algorithm}

\begin{frame}{アルゴリズムの仕組み}
\phantomsection\label{ux30a2ux30ebux30b4ux30eaux30baux30e0ux306eux4ed5ux7d44ux307f}
\begin{enumerate}
\tightlist
\item
  グループ数 \(k\) を決める。
\item
  ランダムに \(k\) 個の中心点（セントロイド）を決める。
\item
  各データを、最も近い中心点のグループに割り当てる。
\item
  グループごとに新しい中心点（平均）を計算する。
\item
  中心点が動かなくなるまで 3 と 4 を繰り返す。
\end{enumerate}
\end{frame}

\section{3.7.4 まとめ}\label{ux307eux3068ux3081}

\begin{frame}[fragile]{このセクションのまとめ}
\phantomsection\label{ux3053ux306eux30bbux30afux30b7ux30e7ux30f3ux306eux307eux3068ux3081}
\begin{itemize}
\tightlist
\item
  \textbf{クラスター分析}:
  データの類似性に基づいて、自動的にグループを生成する。
\item
  \textbf{k-means 法}:
  中心点からの距離を最小化するようにグループを最適化する手法。
\item
  \textbf{政治分析への応用}:
  議員の思想スコアを分類することで、現代の米国議会が政党ごとにきれいに分極化していることを確認できた。
\item
  \textbf{Rの操作}:

  \begin{itemize}
  \tightlist
  \item
    \texttt{matrix()} と \texttt{apply()} による行列計算。
  \item
    \texttt{list()} と \texttt{{[}{[}{]}{]}} による複雑なデータの管理。
  \item
    \texttt{kmeans()} によるアルゴリズムの実行と、\texttt{table()}
    による結果の検証。
  \item
    \texttt{points()} を使った中心点のプロット。
  \end{itemize}
\end{itemize}
\end{frame}
