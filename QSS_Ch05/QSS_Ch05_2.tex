\begin{frame}{社会ネットワーク分析}
\protect\hypertarget{ux793eux4f1aux30cdux30c3ux30c8ux30efux30fcux30afux5206ux6790}{}
\begin{itemize}
\tightlist
\item
  \textbf{ノード} (Nodes/Vertices): 個人や組織などの実体。
\item
  \textbf{エッジ} (Edges/Links):
  ノード間の関係（友人、取引、引用など）。
\item
  \textbf{目標}:
  ネットワークの構造を理解し、中心的な役割を果たすノードを特定する。
\end{itemize}
\end{frame}

\hypertarget{ux307eux3068ux3081}{%
\section{5.2.5 まとめ}\label{ux307eux3068ux3081}}

\begin{frame}{まとめ}
\protect\hypertarget{ux307eux3068ux3081-1}{}
\begin{itemize}
\tightlist
\item
  \textbf{ネットワークデータ}:
  要素間の「つながり」を隣接行列やグラフで表現する。
\item
  \textbf{中心性指標}:

  \begin{itemize}
  \tightlist
  \item
    \textbf{次数}: つながりの多さ。
  \item
    \textbf{近接}: 全体へのアクセスの良さ。
  \item
    \textbf{間}: 仲介役としての重要性。
  \end{itemize}
\item
  \textbf{PageRank}: つながりの「質」を考慮した重要度。
\item
  \textbf{視覚化}: グラフの描画によって直感的な理解を促す。
\end{itemize}
\end{frame}
