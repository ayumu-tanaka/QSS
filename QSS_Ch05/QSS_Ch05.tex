\section{5.4 まとめ}\label{ux307eux3068ux3081}

\begin{frame}[fragile]{第5章のまとめ}
\phantomsection\label{ux7b2c5ux7ae0ux306eux307eux3068ux3081}
\begin{itemize}
\tightlist
\item
  \textbf{テキスト分析}:
  非構造的な文書データから、著者の特徴やトピック（話題）を抽出する。
\item
  \textbf{ネットワーク分析}:
  要素間のつながり（エッジ）と結節点（ノード）を分析し、中心性を測定する。
\item
  \textbf{空間分析}:
  緯度・経度情報を利用して、現象の地理的な広がりを視覚化する。
\item
  Rの操作: \texttt{tm}, \texttt{igraph}, \texttt{maps}
  パッケージ、\texttt{rgb()} による透過色の設定、\texttt{for}
  ループによる地図の着色。
\end{itemize}
\end{frame}
