\section{7.3 仮説検定 (Hypothesis
Testing)}\label{ux4eeeux8aacux691cux5b9a-hypothesis-testing}}

\begin{frame}[fragile]{p値 (p-value)}
\protect\hypertarget{pux5024-p-value}{}
\begin{itemize}
\tightlist
\item
  \textbf{帰無仮説} (\(H_0\)):
  「効果がない（差はゼロである）」という仮定。
\item
  \textbf{p値}: \(H_0\)
  が正しい場合に、観測された結果（またはそれ以上に極端な結果）が得られる確率。
\item
  p値 \textless{} 0.05
  であれば、「統計的に有意」と判断し、帰無仮説を棄却する。
\end{itemize}

\hypertarget{ux307eux3068ux3081}{%
\section{7.5 まとめ}\label{ux307eux3068ux3081}}

\begin{frame}[fragile]{第7章のまとめ}
\protect\hypertarget{ux7b2c7ux7ae0ux306eux307eux3068ux3081}{}
\begin{itemize}
\tightlist
\item
  \textbf{不確実性}: 標本から母集団を推測する際には、常に誤差が伴う。
\item
  \textbf{標準誤差}: 推定値の精度を測る尺度。
\item
  \textbf{信頼区間}: 真の値が含まれる可能性が高い範囲を提示する。
\item
  \textbf{仮説検定}:
  観測された結果が偶然によるものかどうかを確率的に判断する。
\item
  \textbf{統計的有意性}:
  p値が基準（0.05など）を下回るかどうかで判断するが、その解釈には注意が必要。
\item
  Rの操作: \texttt{sd()}, \texttt{qt()}, \texttt{qnorm()},
  \texttt{t.test()}, \texttt{confint()} などを習得した。
\end{itemize}
\end{frame}
