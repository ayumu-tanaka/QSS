\section{4.2 線形回帰 (Linear
Regression)}\label{ux7ddaux5f62ux56deux5e30-linear-regression}}

\begin{frame}{線形回帰の基本概念}
\protect\hypertarget{ux7ddaux5f62ux56deux5e30ux306eux57faux672cux6982ux5ff5}{}
\begin{itemize}
\tightlist
\item
  \textbf{線形回帰}: 結果変数 \(Y\) と説明変数 \(X\)
  の関係を直線で近似する手法。

  \begin{itemize}
  \tightlist
  \item
    数式: \(\hat{Y} = \hat{\alpha} + \hat{\beta} X\)
  \item
    \(\hat{Y}\): 予測値（Fitted value）
  \item
    \(\hat{\alpha}\): 切片（Intercept）
  \item
    \(\hat{\beta}\): 傾き（Slope / 係数）
  \end{itemize}
\item
  \textbf{最小二乗法 (OLS)}: 観測値 \(Y\) と予測値 \(\hat{Y}\)
  の差（残差）の二乗和を最小にする \(\hat{\alpha}\) と \(\hat{\beta}\)
  を選ぶ。
\end{itemize}
\end{frame}

\hypertarget{ux307eux3068ux3081}{%
\section{4.2.5 まとめ}\label{ux307eux3068ux3081}}

\begin{frame}[fragile]{このセクションのまとめ}
\protect\hypertarget{ux3053ux306eux30bbux30afux30b7ux30e7ux30f3ux306eux307eux3068ux3081}{}
\begin{itemize}
\tightlist
\item
  \textbf{線形回帰}: \(Y\) と \(X\) の関係を直線
  \(\hat{Y} = \hat{\alpha} + \hat{\beta} X\) でモデル化する。
\item
  \textbf{最小二乗法}:
  残差の二乗和を最小にすることで、最適な係数を推定する。
\item
  \textbf{データの統合}: \texttt{merge()}
  を使って、複数のソースからのデータを適切に結合する。
\item
  \textbf{適合度の評価}: 決定係数 (\(R^2\))
  や残差プロットを用いて、モデルの良し悪しを判断する。
\item
  \textbf{Rの操作}: \texttt{lm()}, \texttt{coef()}, \texttt{fitted()},
  \texttt{resid()}, \texttt{merge()}, \texttt{summary()\$r.squared}。
\end{itemize}
\end{frame}
