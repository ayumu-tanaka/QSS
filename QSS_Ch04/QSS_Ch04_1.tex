\section{4.1
選挙結果の予測}\label{ux9078ux6319ux7d50ux679cux306eux4e88ux6e2c}

\begin{frame}{予測とは}
\phantomsection\label{ux4e88ux6e2cux3068ux306f}
\begin{itemize}
\tightlist
\item
  \textbf{予測}:
  既存のデータ（訓練データ）からパターンを学び、未知のデータや未来の結果を推測すること。
\item
  社会科学の例：

  \begin{itemize}
  \tightlist
  \item
    世論調査から選挙結果を予測する。
  \item
    経済指標から景気後退を予測する。
  \end{itemize}
\item
  本節では、Rのプログラミングの基礎（ループと条件分岐）を学びながら、2008年米大統領選の予測に取り組みます。
\end{itemize}
\end{frame}

\section{4.1.4 まとめ}\label{ux307eux3068ux3081}

\begin{frame}[fragile]{このセクションのまとめ}
\phantomsection\label{ux3053ux306eux30bbux30afux30b7ux30e7ux30f3ux306eux307eux3068ux3081}
\begin{itemize}
\tightlist
\item
  \textbf{予測のプロセス}:
  データのクリーニング（日付処理など）、サブセットの抽出、要約統計量の計算という一連の流れ。
\item
  \textbf{プログラミングの道具}:

  \begin{itemize}
  \tightlist
  \item
    \texttt{for} ループ:
    大量（今回は51州分）の繰り返し処理を自動化する。
  \item
    \texttt{if} 文: 条件によって処理を分岐させる。
  \end{itemize}
\item
  \textbf{評価}: 実際の値と比較して、誤差（平均誤差や
  RMSE）を算出することで予測精度を客観的に評価する。
\item
  \textbf{Rの操作}: \texttt{as.Date()}, \texttt{for}, \texttt{if},
  \texttt{unique()}, \texttt{min()}, \texttt{mean()}, \texttt{text()}。
\end{itemize}
\end{frame}
