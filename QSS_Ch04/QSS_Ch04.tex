\hypertarget{ux307eux3068ux3081}{%
\section{4.4 まとめ}\label{ux307eux3068ux3081}}

\begin{frame}[fragile]{第4章のまとめ}
\protect\hypertarget{ux7b2c4ux7ae0ux306eux307eux3068ux3081}{}
\begin{itemize}
\tightlist
\item
  \textbf{予測}:
  ループや条件分岐を用いて、大量のデータから将来の結果を推測する。
\item
  \textbf{線形回帰}: \(Y\) と \(X\)
  の関係を直線で近似し、最小二乗法で係数を推定する。
\item
  \textbf{重回帰}: 複数の要因を同時に制御し、因果関係の理解を深める。
\item
  \textbf{RDD}:
  外生的な境界を利用して、観察データから因果効果を抽出する。
\item
  Rの操作: \texttt{for}, \texttt{if}, \texttt{lm()}, \texttt{coef()},
  \texttt{resid()}, \texttt{predict()} などを習得した。
\end{itemize}
\end{frame}
